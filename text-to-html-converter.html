const topics = [

"Marriage, Family & Kinship",

"Culture & Social Change",

"Economic Organization",

"Political Organization",

"Religion, Magic & Ritual",

"Anthropological Theories",

"Caste System",

"Tribe in India",

"Constitutional Safeguards",

"Indian Village Studies",

"Development Anthropology",

"Applied Anthropology",

"Social Stratification & Mobility",

"Indian Social System Overview"

];


const container = document.getElementById('tracker-container');

const chartCanvas = document.getElementById('dashboard-chart');

let dashboardChart;


function updateProgress() {

let completed = 0;

topics.forEach((_, i) => {

if(localStorage.getItem('topic'+i) === 'true') completed++;

});

const percent = Math.round((completed/topics.length)*100);

const bar = document.getElementById('progress-bar');

bar.style.width = percent + '%';

bar.textContent = percent + '%';

return completed;

}


function renderDaily() {

chartCanvas.style.display='none';

container.style.display='flex';

container.innerHTML = '';

topics.forEach((topic, i) => {

const card = document.createElement('div');

card.className = 'tracker-card';

card.innerHTML = `<strong>Day ${i+1}</strong><br>

<input type='checkbox' class='check' id='check${i}'> ${topic}<br>

<textarea placeholder='Notes...'></textarea>`;container.appendChild(card);


const checkbox = card.querySelector('.check');

if(localStorage.getItem('topic'+i) === 'true') checkbox.checked = true;

checkbox.addEventListener('change', () => {

localStorage.setItem('topic'+i, checkbox.checked);

updateProgress();

});

});

updateProgress();

}


function renderWeekly() {

chartCanvas.style.display='none';

container.style.display='flex';

container.innerHTML = '';

let week = 1;

for(let i=0;i<topics.length;i+=7){

const card = document.createElement('div');

card.className = 'tracker-card';

card.innerHTML = `<strong>Week ${week}</strong><br>`;

for(let j=i;j<Math.min(i+7, topics.length);j++){

const checked = localStorage.getItem('topic'+j) === 'true' ? 'checked' : '';

card.innerHTML += `<input type='checkbox' class='check' id='week${j}' ${checked}> Day ${j+1}: ${topics[j]}<br>`;

}

container.appendChild(card);

week++;

}

}


function renderMonthly() {

chartCanvas.style.display='none';

container.style.display='flex';

container.innerHTML = '';

for(let month=1;month<=3;month++){

const card = document.createElement('div');

card.className = 'tracker-card';

card.innerHTML = `<strong>Month ${month}</strong><br>`;

for(let w=1;w<=4;w++){

card.innerHTML += `Week ${w}: <input type='text' placeholder='Notes / progress'><br>`;

}

container.appendChild(card);

}

}


function renderDashboard() {

container.style.display='none';

chartCanvas.style.display='block';

const completed = topics.map((_,i) => localStorage.getItem('topic'+i)==='true'?1:0);

const ctx = chartCanvas.getContext('2d');

if(dashboardChart) dashboardChart.destroy();

dashboardChart = new Chart(ctx, {

type: 'bar',

data: {

labels: topics.map((t,i)=>`Day ${i+1}`),

datasets: [{

label: 'Completion',

data: completed,

backgroundColor: '#4F81BD'

}]

},

options: {

responsive: true,

scales: {

y: {beginAtZero:true, max:1,ticks:{callback: v=>v?'✔':'❌'}}

}

}

});

}


document.getElementById('show-daily').addEventListener('click', renderDaily);

document.getElementById('show-weekly').addEventListener('click', renderWeekly);

document.getElementById('show-monthly').addEventListener('click', renderMonthly);

document.getElementById('show-dashboard').addEventListener('click', renderDashboard);


document.getElementById('toggle-dark').addEventListener('click', () => {

document.body.classList.toggle('dark');

});


document.getElementById('reset-progress').addEventListener('click', () => {

if(confirm('Reset all daily progress?')){

topics.forEach((_, i) => localStorage.removeItem('topic'+i));

renderDaily();

}

});


renderDaily();
